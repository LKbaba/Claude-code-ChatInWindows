// HTML 转义函数，防止 XSS 攻击并正确显示尖括号
function escapeHtml(text) {
    const htmlEntities = {

        '<': '&lt;',

```typescript
function copyMessageContent(messageDiv) {
    const contentDiv = messageDiv.querySelector('.message-content');
    if (contentDiv) {
        // 获取文本内容
        let text = contentDiv.innerText || contentDiv.textContent;

        // 清理多余换行：将 3 个以上连续换行替换为 2 个
        text = text.replace(/\n{3,}/g, '\n\n');
        // 清理首尾空白
        text = text.trim();

        // 复制到剪贴板
        navigator.clipboard.writeText(text).then(() => {
            // 显示反馈（保持原有逻辑）
            const copyBtn = messageDiv.querySelector('.copy-btn');
            const originalHtml = copyBtn.innerHTML;
            copyBtn.innerHTML = '<svg width="12" height="12" viewBox="0 0 24 24" fill="currentColor"><path d="M9 16.17L4.83 12l-1.42 1.41L9 19 21 7l-1.41-1.41z"/></svg>';
            copyBtn.style.color = '#4caf50';

            setTimeout(() => {
                copyBtn.innerHTML = originalHtml;
                copyBtn.style.color = '';
            }, 1000);
        }).catch(err => {
            console.error('Failed to copy message:', err);
        });
    }
}
```

**文件内容**:

```typescript
/**
 * ClaudeConfigService - 管理 ~/.claude/settings.json 配置文件
 * 用于写入官方 language 配置
 */

import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import { debugLog } from './DebugLogger';

// 语言代码到官方语言名称的映射
const OFFICIAL_LANGUAGE_NAMES: { [key: string]: string } = {
    'zh': 'chinese',
    'es': 'spanish',
    'ar': 'arabic',
    'fr': 'french',
    'de': 'german',
    'ja': 'japanese',
    'ko': 'korean'
};

/**
 * 获取 Claude settings.json 文件路径
 */
function getClaudeSettingsPath(): string {
    return path.join(os.homedir(), '.claude', 'settings.json');
}

/**
 * 读取 Claude settings.json 文件
 */
async function readClaudeSettings(): Promise<any> {
    const settingsPath = getClaudeSettingsPath();
    try {
        const content = await fs.promises.readFile(settingsPath, 'utf-8');
        return JSON.parse(content);
    } catch (error) {
        // 文件不存在或解析失败，返回空对象
        debugLog('ClaudeConfigService', 'Failed to read settings.json, using empty object', { error });
        return {};
    }
}

/**
 * 写入 Claude settings.json 文件
 */
async function writeClaudeSettings(settings: any): Promise<void> {
    const settingsPath = getClaudeSettingsPath();

    // 确保目录存在
    const settingsDir = path.dirname(settingsPath);
    await fs.promises.mkdir(settingsDir, { recursive: true });

    // 写入文件
    await fs.promises.writeFile(settingsPath, JSON.stringify(settings, null, 2), 'utf-8');
    debugLog('ClaudeConfigService', 'Settings written successfully', { path: settingsPath });
}

/**
 * 设置 Claude 的 language 配置
 * @param languageCode 语言代码（如 'zh', 'ja'）
 */
export async function writeClaudeLanguageSetting(languageCode: string): Promise<void> {
    try {
        const settings = await readClaudeSettings();

        // 转换为官方语言名称
        const officialName = OFFICIAL_LANGUAGE_NAMES[languageCode] || languageCode;
        settings.language = officialName;

        await writeClaudeSettings(settings);
        debugLog('ClaudeConfigService', 'Language setting updated', {
            languageCode,
            officialName
        });
    } catch (error) {
        debugLog('ClaudeConfigService:ERROR', 'Failed to write language setting', { error });
        throw error;
    }
}

/**
 * 移除 Claude 的 language 配置
 */
export async function removeClaudeLanguageSetting(): Promise<void> {
    try {
        const settings = await readClaudeSettings();

        if (settings.language) {
            delete settings.language;
            await writeClaudeSettings(settings);
            debugLog('ClaudeConfigService', 'Language setting removed');
        }
    } catch (error) {
        debugLog('ClaudeConfigService:ERROR', 'Failed to remove language setting', { error });
        // 不抛出错误，静默失败
    }
}

/**
 * 获取当前 Claude 的 language 配置
 */
export async function getClaudeLanguageSetting(): Promise<string | null> {
    try {
        const settings = await readClaudeSettings();
        return settings.language || null;
    } catch {
        return null;
    }
}
```
